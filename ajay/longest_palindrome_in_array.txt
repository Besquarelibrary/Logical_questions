#include<stdio.h>
int ispalindrome(int n){
     int rev=0, temp = n;

     while(temp){
          int rem = temp%10;
          rev = rev*10 + rem;
          temp /= 10;
     }
     
      if(n==rev)
        return 1;
      else
        return 0;
}

int main()
{
    int arr[] = {121,2551552,6116,07,1001},k=0;
    int n = sizeof(arr)/sizeof(arr[0]);
  
    for(int i=0; i<n; i++)
    {
             if(ispalindrome(arr[i]))
             {
                if(arr[i]>k)
                {
                  k=arr[i];      
                }
             }
    }
    printf("%d ",k);
}
