#include <stdio.h>
#include <string.h>
int main() {
    char s[] = "racecar";
    int freq[256] = {0};
    int len = strlen(s);
    for (int i = 0; i < len; i++) {
        freq[s[i]]++;
    }
    int odd_count = 0;
    for (int i = 0; i < 256; i++) 
    {
        if (freq[i] % 2 != 0) {
            odd_count++;
        }
    }
    if (len % 2 == 0 && odd_count == 0 || odd_count == 1) 
    {
        printf("it''s a palindrome\n");
    }
    else
	printf("it's not a palindrome\n");
    return 0;
}
