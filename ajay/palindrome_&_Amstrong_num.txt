void prime_check(int num)
{
    int flag=0,i;
    
   if (num == 0 || num == 1)
      printf("%d is not a prime number\n", num);
   else 
   {
      for (i = 2; i <= num/2; ++i) 
      {
         if (num % i == 0) 
         {
            flag = 1;
            break;
         }
      }
      if (flag == 0)
      {
         printf("%d is a prime number\n", num);
      }
      else
         printf("%d is not a prime number\n", num);
   }
}

void amstrong_check(int num)
{
    int originalNum,sum=0,n=0,rem; 
    
    originalNum = num;
    while (originalNum != 0) 
    {
      originalNum /= 10;
      ++n;
    }
    
   originalNum = num;
   while (originalNum != 0) 
   {
      rem = originalNum % 10;
      sum += pow(rem, n);
      originalNum /= 10;
   }
   if (sum == num)
   {
      printf("%d is an Armstrong number\n", num);
   }
   else
      printf("%d is not an Armstrong number\n", num);
}
