#include <stdio.h>
void check_perfect(int);
int main() 
{
    int n;
    printf("enter a  number: ");
    scanf("%d",&n);
    check_perfect(n);
    return 0;
}

void check_perfect(int n)
{
    int sum=0;
    for(int i=1; i<n; i++)
    {
        if(n%i==0)
        {
            sum+=i;
        }
    }
    if(sum==n)
    printf("perfect number\n");
    else 
    printf("not a perfect number\n");
}
