#include<stdio.h>
int main()
{
int arr[]={0,1,2,1,0,2,1,1,0};
int n=sizeof(arr)/sizeof(arr[0]);
int low = 0, mid = 0, high = n - 1,i; 
    while (mid <= high) {
        switch (arr[mid]) {
            case 0:
                int temp = arr[low];
                arr[low] = arr[mid];
                arr[mid] = temp;
                low++;
                mid++;
                break;
            case 1:
                mid++;
                break;
            case 2:
                temp = arr[mid];
                arr[mid] = arr[high];
                arr[high] = temp;
                high--;
                break;
        }
    }
    printf("the sorted array is ");
    for(i=0; i<n; i++)
    {
        printf("%d ",arr[i]);
    }
}
