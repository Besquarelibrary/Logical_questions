#include <stdio.h>

int equilibrium(int arr[], int n) 
{
   int a = 0, b = 0;
  // adding all elements
   for (int i = 0; i < n; i++) 
   {
      a += arr[i];
   }

   for (int i = 0; i < n; i++) 
   {
      a -= arr[i]; //removing each element from sum
      if (b == a) //checking if a==b
      {
         return arr[i]; // returning that index value
      }
      b += arr[i]; // adding again
   }

   return -1;
}

int main() 
{
   int arr[] = {1,5,4,5,1};
   int n = sizeof(arr) / sizeof(arr[0]);

   int x = equilibrium(arr, n);

   if (x == -1) 
   {
      printf("No equilibrium index found\n");
   } 
   else 
   {
      printf("Equilibrium number is %d\n",x);
   }
}
