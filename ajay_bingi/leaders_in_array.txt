#include <stdio.h>

void printLeaders(int arr[], int size)
{
    int leaders[size], num_leaders = 0;

    for (int i = 0; i < size; i++) 
    {
        int is_leader = 1;
        for (int j = i + 1; j < size; j++) 
        {
            if (arr[j] > arr[i]) {
                is_leader = 0;
                break;
            }
        }
        if (is_leader) 
        {
            leaders[num_leaders++] = arr[i];
        }
    }

    printf("Leaders: ");
    for (int i = 0; i < num_leaders; i++) 
    {
        printf("%d ", leaders[i]);
    }
    printf("\n");
}

int main()
{
    int arr[] = {1,7,9,8,3,6};
    int size = sizeof(arr)/sizeof(arr[0]);

    printLeaders(arr, size);

    return 0;
}
