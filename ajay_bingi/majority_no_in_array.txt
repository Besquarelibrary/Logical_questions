#include <stdio.h>
int main() 
{
    int arr[]={1,1,2,2,2,2,2,2,2,3,4,5,5,6};
    int len=sizeof(arr)/sizeof(arr[0]); // finding length or array
    int half_len=len/2; // dividing the length of array
    int temp[len],x=0; // temporary array 
    for(int i=0; i<len; i++) 
    {
        temp[arr[i]]=0;  //assigning zero's 
    }
    for(int j=0; j<len; j++)
    {
        temp[arr[j]]++; //incrementing the values of temp array
    }
    
    for(int k=0; k<len; k++)
    {   
        if(temp[k]>=half_len) //checking the value is greater than or equal to hal_length of array 
        {
            x++;
            printf("%d is the majority one among all elements",k);
        }
    }
    if(x==0)
    printf("no majority number in a given arrray\n");
    
}
