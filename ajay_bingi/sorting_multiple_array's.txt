#include<stdio.h>
int main()
{
    int arr1[]={0,3,5,9,6}, arr2[]={2,7,1,8,4};
    int l1=sizeof(arr1)/sizeof(arr1[0]);
    int l2=sizeof(arr2)/sizeof(arr2[0]);
    int temp[l1+l2],i,j,m,k,a,b;
    for(i=0; i<l1; i++)
    {
        temp[i]=arr1[i];
    }
    
    for(j=0; j<l2; j++,i++)
    {
        temp[i]=arr2[j];
    }
    int l3=sizeof(temp)/sizeof(temp[0]);
    //using bubble_sort 
    for(k=0; k<l3; k++)
    {
        for(m=k+1; m<l3; m++)
        {
            if(temp[k]>temp[m])
            {
            int t=temp[m];
            temp[m]=temp[k];
            temp[k]=t;
            }
        }
    }
    for(a=0; a<l1; a++)
    {
        arr1[a]=temp[a];
    }
    for(b=0; b<l2; b++,a++)
    {
        arr2[b]=temp[a];
    }
    //printing the array's
    for(a=0; a<l1; a++)
    {
        printf("%d ",arr1[a]);
    }
    printf("\n");
    for(a=0; a<l2; a++)
    {
        printf("%d ",arr2[a]);
    }
    
}
